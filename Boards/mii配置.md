
- 写入前复位，后续寄存器写入才有效
mii write 0x00 0x1e 0x0000
mii write 0x00 0x00 0x8000



- normal
mii read 0x00 0x00# 读取0:5保留位，写入时需保持一致
mii write 0x00 0x00 0x1100# 全双工，10Mbps，启用自动协商，loopback disable
mii write 0x00 0x00 0x3100# 全双工，100Mbps，启用自动协商，loopback disable
mii write 0x00 0x00 0x1140# 全双工，1000Mbps，启用自动协商，loopback disable

mii read 0x00 0x00
mii write 0x00 0x00 0x1100
mii write 0x00 0x00 0x3100
mii write 0x00 0x00 0x1140

mii read 0x00 0x09# 0:7位写入时需保持一致
mii write 0x00 0x09 0x0200  # 默认值：启用1000Mbps全双工协商，半双工disable

mii read 0x00 0x09
mii write 0x00 0x09 0x0200

- loopback
mii write 0x00 0x00 0x7100# 全双工，100Mbps，启用自动协商，loopback enable
mii write 0x00 0x00 0x7100
